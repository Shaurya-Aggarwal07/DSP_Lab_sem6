n = 0:50;
x = cos(pi*n/3).*(n>=0);
y = zeros(size(n));

y_minus1 = -2;
y_minus2 = -3;
x_minus1 = 1;
x_minus2 = 1;

for k = 1:length(n)
    if k==1 % n==0
        y(k) = (1/3)*(x(k)+x_minus1+x_minus2) + 0.95*y_minus1 - 0.9025*y_minus2;
    elseif k==2 % n==1
        y(k) = (1/3)*(x(k)+x(k-1)+x_minus1) + 0.95*y(k-1) - 0.9025*y_minus1;
    else
        y(k) = (1/3)*(x(k)+x(k-1)+x(k-2)) + 0.95*y(k-1) - 0.9025*y(k-2);
    end
end

stem(n,y,'filled');
grid on;
